قد تكون هذه الحكة وقلة الشعر بسبب وجود القشرة، قشـــرة الشعـــــــر Dandruff يعتبر قشر فروة شعر الرأس من أكثر المشاكل الجلدية المتكررة التي تورق الكثير من الناس . أن فـروة رأس الإنسان تغطيهــا طبقــة الخــلايـــا الكراتينيــة ( Keratin layer ) تتجــدد بصفــة مستمــرة ، ويعنــى ذلك أن هنــاك خلايــا جلدية تكبـر فتمــوت، و تسقـط لتحــل محلهــا خلايــا أخــرى شـابـة سليمـة قـوية و حيــة . في فـروة الـرأس يـوجــد الشعــر ، و هذا الشعــر يعــوق الخـلايــا الميتــة مـن السقــوط ، ممــا يؤدى إلى تـراكــم وتجمــع هذه الخلايا الميتة في شكــل صفـائح رقيقــة صغيــرة في الحجــم تأخــذ شكــل القشــور التي يعـرفهـا الجميــع. وتلتصــق هـذه القشــور بعضهـا ببعض وذلك نتيجــة لـوجــود إفـرازات الغـدد الدهنيــة بفــروة الرأس . أن فروة الرأس تتميز بأنها غنية بالغـدد الدهنيــة ، وكـل هـذا يؤدى إلى تجمــع و التصــاق الخلايا الميتـة و ازديادهـا في الحجــم . وقد يختلف ظهور قشـر الشعــر من شخـص إلى آخر ، أو يظهــر عنــد بعـــض النـاس دون غيــرهــم ،، وتلعــب الهـرمونــات المـوجـودة في الجســم دورا هـامـا في حدوث ذلك . كمـا أن ذوى الشعـور الطـويلة أكثـر عرضـة للإصابة به . و الأسباب التي تـؤدى إلى ظهــور قشـر الرأس ، أو تـزيـد منـه ، عديدة أهمهـا : الاستعـداد الشخصي ، و البشـرة الدهنيـة، و طـول الشعـر ، وتـرك الشعـر بدون غسـيل مـدة طويلة . كما أن استخـدام الصـابـون أو الشـامبـو الذي يحتـوى على مـواد مهيجـة للجلـد قد يؤدى إلى زيادة نشـاط الغـدد الدهنيـة الموجـودة بفروة الرأس وقد يكون هذا النشـاط الزائـد من غير سبب .. أي يتعلـق بطبيعـة تكـوين الجسـم .. كمـا أن هناك عـوامـل تسـاعـد على الإصابة بقشـر الشعـر مثل الإفراط في تنـاول الأطعمة الحارة و البهـارات و التـوابـل و الشيكـولاتــة و المشـروبـات السـاخنة جـدا .. و التي تؤدى لزيـادة تـوارد الـدم للغــدد الدهنيــة ، و زيـادة نشـاطهـا في إفراز الدهــون ، و كذلك يؤدى تمشيـط الشعــر بعنـف و حــك فروة الـرأس بالمشط إلى زيـادة القشـور و فرصــة حـدوث التهـاب دهني بفـروة الرأس . ويمكن أن تؤدى الأدوات الكهربائية المستخدمة لتجفيف وتصفيف الشعر مثل السيشوار إلى الضرر في جلد فروة الرأس، وتسبيب القشرة، خاصة فى حالة عدم استعمالها بطريقة صحيحة، أو إذا أكثر من استعمالها، ويمكن أن تسبب حبوب منع الحمل ظهور القشرة في جلدة الرأس، لأنها تؤثر في معدل نمو خلايا الجلد. كما أن زيادة إفراز العرق الناتج عن ارتداء القبعات الضيقة وأغطية الرأس تزيد من إفرازات العرق في جلدة الرأس مما يزيد من ظهور مشكلة دهنية تسبب القشرة. يعد أصحاب البشرة الدهنية أكثر إصابة بقشرة الشعر، ويرجع ذلك إلى كثرة إفرازهم للدهون التى تساعد على تكوين طبقات من القشور بجلد وأصحاب الشعر الطويل أكثر عرضة للإصابة بقشرة الرأس، حيث نجد فى بعض الحالات أن الشعر، خاصة إذا كان طويلا يعوق سقوط هذه الخلايا الميتة، مما يؤدى إلى تراكمها وتجمعها فى شكل صفائح رقيقة صغيرة في الحجم تأخذ شكل القشور . وهي غيـر معدية إذا لـم تكـن مصحوبة بالتهـابات في فروة الرأس ،. وقــد أكد كثيــر من البـاحثيــن وجـود علاقــة وطيدة بيـن البشــرة الدهنيــة وبيـن قشــر الشعـر ، و يلاحـظ كذلك أن الحـالة النفسيـة لها دور في هذا المـوضوع ، حيث تزيـد مشكلة قشـر الشعر مع القلق و التوتـر و العصبيــة بينمـا يمكن لكثير من الحالات من الانحصــار و التحسن مـع الاستقـرار النفسي . وقـد أجريت عـدة بحوث مقارنه بين فئات العمــر و وجد انـه كلما تقـدم العمـر بالإنسان كلمـا قل حدوث القشرة وقل إفراز الغدد الدهنيـة . وفي بعـض ألا حيان قـد يكــون ظهــور قشــر الشعــر مصحـوبـا بالتهــاب في فروة الـرأس وتدهنهــا وتسمى هذه الحـالة المـرضيــة بالاكـزيمــا الدهنيــة ( Seborrheic dermatitis ) وهى نـوع مـن أنواع الاكـزيمـا التي تصيب كل من الرأس ، الجبهة ، الوجه ، جانبي الأنف و الذقن ، الرموش ، داخل الأذن ، الصدر و ثنايا الجلـــد. وتتميـز هذه الأكزيما بظهور إفرازات صفـراء اللـون وتكــون هذه الحالات مصحوبة بالتهـابات و بحكة شديدة.. وتؤدى هذه الحكـة إلى إصابة فروة الرأس ببعـض الميكـروبــات التي تؤدى الى ظهــور دمـامـل ( Boils ) .. أو بثــور صـديدية صغيــرة ( Pustules ) ...وهذا مـا يمكن أن نسميــه : التهــاب البشــرة الدهني . وقـد يسبب الإهمال في عــلاج هذه الاكزيمــا الى حدوث حكــة شديدة بالرأس و انتشــار العدوى ، وتكـرار هذه الاكـزيمـا يؤدى الى حدوث سقــوط كثيف خصـوصـا في مقــدمــة الرأس و تؤدى في النهاية الى صلع رجالي مبكـر في الذكــور ( Androgenic Alopecia ) ، و لكن في الإناث يخـف الشعــر ، لكن لا يحدث صلــع كامــل وهذا مـا أكدته كثيــر من الدراســات و الأبحاث في هذا المجــال . وقد اكتشف حديثـا و جـود علاقــة بين ظهـور قشـر الرأس و وجـود فطـر خميري ينمـو بصفـة طبيعيـة على سطح الجلد يسمـى " بيترسـبورم "، فكلمـا ازدادت أعـداد ذلك الفطـر كان القشـر أكثــر كثـافة .. وهنــاك أسباب كثيـرة أخرى لظـهـور قشـر الرأس منهـا : 1- الصـدفيـة و قشــورهـا ( Psoriasis ) : الصـدفية مرض جلدي مزمن يصيب الجسـم ونسبة حدوثه 3% من الناس تقريبا ، ويتميـز هذا المـرض بظهور طفـح جلدي ملتهب مزمن محمــر مغطى بقشـور بيضـاء فضيــة اللون في مناطق متعددة بالجسم و الرأس ، وتعتبــر الصـدفية من الأمراض الجلدية غير المعـدية ،والذي يحدث التسـاقط هــو وجود الحكة الشديدة المصاحبة للصـدفية بالرأس . 2- الذئبــة الحمـراء ( Lupus erythematosus ) : وهــو مرض جلدي نادر ونسبــة حدوثه اكثر في السيدات ، ويصيب الوجـه و المناطق المعرضــة لأشعـة الشمـس ويظهـر في شكــل الفـراشـة و حافتــه محــددة و نشطــة و محمــرة ووسطــه اقـل نشـاطـا ويظهر ضمــور ملحـوظ عن بقيـة الجلد ، وتتقيـح مسـام الشعـر وتظهـر فيهـا قشــور صفراء صغيرة من فوهــات المســام و يصيــب أيضا الرأس و يسبب إصابة ممـاثلـة لإصابة الوجــه مـع ظهــور الندب و القشــور و يؤدى في النهايـة الى حدوث صلـع دائــم في المنطقــة المصــابة . 3- الأمراض الفطرية بالرأس Fungal Infection ) : الأمراض الفطرية بالرأس عديدة منها ما يسمى بالقراع البلدي و القراع العسلي ، ويظهر في شكل دوائر حلقية مصفرة ، وهناك نوع آخر من القرع العسـلي و هو النـوع القشــري وبــه تظهـر منطقـة ملتهبـة بهـا قشـور جافـة على شكل دوائر حلقية . وهناك نوع شائـع من القـرع يسمى القـراع الإنجليزي ( Tinea Capitis ) ( القـوباء الحلقيـة) وفيـه تظهـر منطقــة ملتهبــة محـددة الحـافة و يتقصف فيهـا الشعـر في شكـل جذوع و تظهر بهـا قشــور صغيـرة بيضــاء . و القـراع مـرض معد وخصـوصـا للأطفال ، وقــد تنتـج العـدوى من طفـل لآخر أو من بعـض الحيـوانـات كالقطط و الكـلاب الى الأطفال، أو من الحـلاقيـن إذا استعمـل الحـلاق الأدوات التي حلـق بهـا للطفــل المصــاب فبـل تعقيمهــا . 4- أمراض أخرى : قد يسبب نقـص فيتـامين ( أ ) من جسم الإنسان الى جفــاف البشـرة ، وبروز بصيـلات الشعــر ، و قشــور بالـرأس و الجســم ، و إصابات أخرى تشـابه إصابات حب الشبـاب و تضخــم البشــرة باليــدين و القدمين ، و إصابات محمــرة متكسرة تشــابه الى حـد مـا مـرض الصدفية المـزمن ( Pityriasis Rubra Pilaris ) . و هناك الاكزيمــا القشـرية التي تنتـج من أمراض زيادة الحسـاسية ( Discoid eczema - Neurodermatitis )، وهي من الأمراض التي تصيـب أي مكـان من الجســم و تحـدث حكـة شـديدة . عـلاج قشــر الشعــر : في البدايــة يجب أن نسأل لمــاذا نعــالج قشــر الرأس ، وما هي المضـاعفـات التي سوف تنتج من عــدم علاج هــذه القشــرة عند كـل من الرجل و المرأة ؟ إن الشكوى الرئيسيــة عند كــل من يشتكى من قشـر الرأس تكــون من تســاقط القشــر من الرأس إلى الملابــس ممـا يعطى منظـرا غير محبب ، و كذلك الرغبــة في حك جلد فروة الرأس باستمرار أمام الناس ، و الإحساس بحرقـان متكـرر في فروة الرأس ، ويرجــع سبـب هذه الأعراض كما ذكرنـا الى البشرة الدهنيــة المصاحبـة للقشـور ، ممـا يؤدي الى تهيج أعصاب فروة الرأس و الإحساس بهذا الحرقـان و الهرش الشديد عند هؤلاء الناس . و السبب الأخر هــو كما ذكرنا في السابق من ارتباط وجـود قشر الرأس و زيـادة سقـوط الشعــر عنـد كثير من الناس ، فتسـاقط بعض الشعيـرات في حالـة وجـود قشـر في الرأس لبعـض الإناث يكـون نتيجــة للحكـة التي تصـاحب ظهــور هـذه القشــور ، وفى كثير من الأحيان يكون قشر الشعر مصحوبا بتغيـر في فروة الرأس وهذا يعنى وجود مرض جلدي بفروة الرأس في بـدايــة عـلاج قشـر الرأس يستحسن إعطاء المـريض جرعــات من فيتامين ( أ ) وكذلك فيتامين ( ب ) المـركب . وكذلك منشـطات الكبــد و هضـــم الدهنيــات مثل حبوب دانزن حبة بعد كل وجبة. ويمنـع تعـاطى المــواد الدهنيــة و يعـالج عســر الهضـم ، ويراعى الإقلال من السكر أو استبدالـه بعسل النحـل . أن اسهـل و ابسـط طريقــة للوقـايـة من قشـر فروة الـرأس هي تجنب الأسباب التي تزيـد من قشــر الشعـر ، وكذلك فان تقصيـر الشعر في الرجال من أهم العـوامل التي تسـاعـد على اختفـاء القشـور ، إذ أن تقصيـر الشعـر يمنعـه من احتجـاز خلايــا البشـرة و يسـاعد على تسـاقطهـا بسهولـة . ويراعى تكـرار غسيل فروة الرأس بالمــاء الفاتر ، وليس السـاخن واستخدام الصــابون العادي الذي لا يحتوى على مـواد كيمـاويـة ويؤدى الى إزالة هـذه الطبقـة الدهنيـة التي تؤدي الى قشـر الرأس ، فغسل الشعر يسـاعـد على إزالة هذه الطبقـة الدهنية، ويشتكى كثير من الناس من زيادة قشر الشعر خلال فصـل الشتـاء عنـه في الصيف ، ويرجع ذلك الى ارتفاع درجة الحـرارة في الصيف ممـا يؤدى الى زيـادة إفراز العــرق الذي بدوره يسبب سيولـة الدهـون ، و بالتالي عــدم التصـاقهـا بالبشـرة ، مما يقلل احتمـال تكون قشـرة الرأس . أمـا في الشتاء فان انخفـاض درجـة الحـرارة ، وقلـة إفراز العـرق يؤديـان الى تجمـد الدهـون و زيـادة التصـاقهـا بخلايـا البشـرة ممـا يسهـل تجمعهــا ، و يسـاعـد على التصـاق الخلايـا و تجمعهـا على شكل قشـور بالرأس ، ويجب أزالتها بالشـامبـو . أن استعمـال الشـامبو باستمرار يؤدى الى إزالة القشــور، إذ أن الشـامبـو فيـه بعض الكيمـاويـات المنظفـة التي تســاعد على إزالة القشــور و الأتـربـة و الأوساخ التي قـد تكـون عالقـة بفروة الرأس أو الشعر . وهنــاك أنـواع كثيـرة من الشـامبـو أو المستحلب للتخلص من قشـر الشعـر مثل شامبوهـات تحتوى على مادة الزنـك مثل شامبو بايبيترون و أخرى تحتـوى على الكبـريت و حمض السالي سيلك وافضل أنواع الشامبوهات لمعالجة القشرة هو شامبو جلايتون مرتين في الأسبوع ثم يستعاض عنه بشامبو هيربال جلو، وشامبو سَِلسَن. أو شامبو هيد آند شولدر أو شامبو كلير كلها ممتازة ومجربة في علاج القشرة.وكذلك التي تحتوى على مواد مضادة للفطريات و الميكروبات مثل الكيتوكنازول أو شامبو نيزورال أو شامبو سيبورال ويستخـدم بعمـل تدليك لفروة الرأس بكميــة من المستحضـر لمـدة خمس دقائق ، ويشـطف بعـدهـا بالمـاء الفاتـر.. ويكـرر ذلك أسبوعيا عدة مرات حسب شدة الحالة . كـذلك يمكن استخـدام الزيت الدافئ لفـروة الـرأس في المســاء فبـل النــوم بكميــة من زيـت الـزيتـون ، أو زيـت اللـوز ، أو كـريم نبــات الصبــار .. مـع عمـل تدليك لفـروة الرأس .. ويلف الشعـر بفـوطة دافئة، ثـم يغســل في الصبــاح بالمـاء العادي . ويمكن
