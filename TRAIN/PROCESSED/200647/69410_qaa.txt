يبدو أن السبب وجود القشرة Dandruffفي رأسك ويعتبر قشر فروة شعر الرأس من أكثر المشاكل الجلدية المتكررة التي تورق الكثير من الناس . أن فـروة رأس الإنسان تغطيهــا طبقــة الخــلايـــا الكراتينيــة ( Keratin layer ) تتجــدد بصفــة مستمــرة ويعنــى ذلك أن هنــاك خلايــا جلدية تكبـر فتمــوت و تسقـط لتحــل محلهــا خلايــا أخــرى شـابـة سليمـة قـوية و حيــة . في فـروة الـرأس يـوجــد الشعــر و هذا الشعــر يعــوق الخـلايــا الميتــة مـن السقــوط ممــا يؤدى إلى تـراكــم وتجمــع هذه الخلايا الميتة في شكــل صفـائح رقيقــة صغيــرة في الحجــم تأخــذ شكــل القشــور التي يعـر فهـا الجميــع. وتلتصــق هـذه القشــور بعضهـا ببعض وذلك نتيجــة لـوجــود إفـرازات الغـدد الدهنيــة بفــروة الرأس يستحسن إعطاء المـريض جرعــات من فيتامين ( أ ) وكذلك فيتامين ( ب ) المـركب . وكذلك منشـطات الكبــد و هضـــم الدهنيــات مثل حبوب دانزن حبة بعد كل وجبة. ويمنـع تعـاطى المــواد الدهنيــة و يعـالج عســر الهضـم ويراعى الإقلال من السكر أو استبدالـه بعسل النحـل . أن اسهـل و ابسـط طريقــة للوقـايـة من قشـر فروة الـرأس هي تجنب الأسباب التي تزيـد من قشــر الشعـر وكذلك فان تقصيـر الشعر في الرجال من أهم العـوامل التي تسـاعـد على اختفـاء القشـور إذ أن تقصي? ?ر الشعـر يمنعـه من احتجـاز خلايــا البشـرة و يسـاعد على تسـاقطهـا بسهولـة . ويراعى تكـرار غسيل فروة الرأس بالمــاء الفاتر وليس السـاخن واستخدام الصــابون العادي الذي لا يحتوى على مـواد كيمـاويـة ويؤدى الى إزالة هـذه الطبقـة الدهنيـة التي تؤدي الى قشـر الرأس
